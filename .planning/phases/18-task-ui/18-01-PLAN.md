    ```typescript
      list: (projectKey: string, archived = false) =>

        request<GsdTask>(`/gsd/projects/${encodeURIComponent(projectKey)}/tasks/${taskId}`, {
          method: "PATCH",
          body: JSON.stringify(patch),
        }),
    },
    ```
