<objective>
Create an Express middleware (`createAgentProxy`) that transparently proxies GET requests for agent data routes to the upstream server at `GSD_DATA_URL`, and mount it in `server/index.js` before the existing SQLite-backed routers.

Purpose: The Railway-hosted server has an empty SQLite database because Claude Code hooks POST to localhost. By proxying GET requests through the same `GSD_DATA_URL` tunnel already used for GSD file routes, the Railway dashboard shows real session data from the local machine.
Output: `server/routes/proxy.js` (new file) + minimal wiring changes in `server/index.js`.
</objective>

Create `server/routes/proxy.js` with the following implementation:

```js
const AGENT_PREFIXES = [
  '/api/sessions',
  '/api/agents',
  '/api/events',
  '/api/stats',
  '/api/analytics',
];

function createAgentProxy(gsdDataUrl) {
  if (!gsdDataUrl) return (_req, _res, next) => next();

  return async function agentProxy(req, res, next) {
    if (req.method !== 'GET') return next();
    if (!AGENT_PREFIXES.some((p) => req.path.startsWith(p))) return next();

    const search = req.url.includes('?') ? req.url.slice(req.url.indexOf('?')) : '';
    const upstreamUrl = `${gsdDataUrl}${req.path}${search}`;

    try {
      const upstream = await fetch(upstreamUrl, { signal: AbortSignal.timeout(10000) });
      const contentType = upstream.headers.get('content-type') || 'application/json';
      const body = await upstream.text();
      res.status(upstream.status).set('Content-Type', contentType).send(body);
    } catch (err) {
      res.status(502).json({ error: 'Failed to reach agent data source', detail: err.message });
    }
  };
}

module.exports = { createAgentProxy };
```
