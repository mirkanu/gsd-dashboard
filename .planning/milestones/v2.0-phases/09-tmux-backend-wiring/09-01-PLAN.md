  <behavior>
    isTmuxSessionActive(sessionName):
    - Returns false immediately if sessionName is falsy (null, undefined, empty string)
    - Calls execFileSync('tmux', ['has-session', '-t', sessionName]) — throws if session not found
    - Returns true if execFileSync succeeds (exit code 0)
    - Returns false on any thrown error (session not found, tmux not installed, etc.)
    - Never throws — always returns boolean
    - Exported as module.exports = { isTmuxSessionActive }
  </behavior>
  <action>
Create `server/gsd/tmux.js` with:

```js
'use strict';

const { execFileSync } = require('child_process');

/**
 * Check whether a named tmux session exists and is running.
 * Returns false for falsy input or any error (tmux not found, session absent, etc.).
 * Never throws.
 */
function isTmuxSessionActive(sessionName) {
  if (!sessionName) return false;
  try {
    execFileSync('tmux', ['has-session', '-t', sessionName], { stdio: 'ignore' });
    return true;
  } catch {
    return false;
  }
}

module.exports = { isTmuxSessionActive };
```
